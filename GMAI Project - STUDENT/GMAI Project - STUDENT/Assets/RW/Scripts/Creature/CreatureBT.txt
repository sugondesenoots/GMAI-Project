tree("Root") 
    repeat
        fallback
            tree("CreatureIdle") 
            tree("CreatureRoam") 
            tree("CreatureRetreat")

tree("CreatureIdle")
    sequence
        IsIdleState  
        CheckForOthersIdle
        SwitchToRoam 
         
tree("CreatureRoam")
    sequence
        IsRoamState  
        Roam
        CheckForOthersRoam
        SwitchToRetreatOrFollow
         
tree("CreatureRetreat")
    sequence
        IsRetreatState 
        Retreat   
        SwitchToIdle
