tree("Root") 
    repeat
        fallback
            tree("CreatureIdle") 
            tree("CreatureRoam") 
            tree("CreatureRetreat") 
            tree("CreatureFollowFood") 
            tree("CreatureEat")

tree("CreatureIdle")
    sequence
        IsIdleState  
        CheckForOthersIdle
        SwitchToRoam 
         
tree("CreatureRoam")
    sequence
        IsRoamState  
        Roam
        CheckForOthersRoam
        SwitchToRetreatOrFollow
         
tree("CreatureRetreat")
    sequence
        IsRetreatState 
        Retreat   
        SwitchToIdle 
         
tree("CreatureFollowFood") 
    fallback
        sequence
            IsFollowFoodState
            FollowFood
        sequence 
            IsFollowFoodState
            SwitchToRoamOrEat 
         
tree("CreatureEat")
    sequence
        IsEatState  
        Eating 
        SwitchToRoamEat
